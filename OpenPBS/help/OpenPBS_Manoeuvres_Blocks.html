block RollingMax "Find max value of n signals during simulation" extends Modelica.Blocks.Interfaces.BlockIcon; parameter Integer n1=1 "Number of rows in input matrix"; parameter Integer n2=1 "Number of columns in input matrix"; Real[n1,n2] umax; Modelica.Blocks.Interfaces.RealInput u[n1,n2](stateSelect=StateSelect.never); Modelica.Blocks.Interfaces.RealOutput y; equation y = max(umax); for i in 1:n1 loop for j in 1:n2 loop when der(u[i,j])<0 then umax[i,j] = max(u[i,j],pre(umax[i,j])); end when; end for; end for; end RollingMax;

block Curvature "Curvature at each axle" extends Modelica.Blocks.Interfaces.BlockIcon; parameter Integer nu=3 "Number of units"; parameter Integer na=2 "Max number of axles per unit"; Modelica.Blocks.Interfaces.RealInput vx[nu] "vx at c.o.g."; Modelica.Blocks.Interfaces.RealInput vy[nu] "vy at c.o.g."; Modelica.Blocks.Interfaces.RealInput wz[nu] "wz at c.o.g."; parameter Modelica.SIunits.Length[nu,na] Lcog "Axle positions relative c.g."; Modelica.Blocks.Interfaces.RealOutput y[nu,na] "Connector of Real output signals"; equation for i in 1:nu loop for j in 1:na loop y[i,j] = wz[i]/sqrt((vy[i]+Lcog[i,j]*wz[i])^2+vx[i]^2); end for; end for; end Curvature;
